create schema estabelecimento;


use estabelecimento;

CREATE TABLE Cliente (
id INT PRIMARY KEY auto_increment,
nome VARCHAR(30),
telefone  VARCHAR(15),
email varchar(40)
);

CREATE TABLE Pedido (
id INT primary key auto_increment,
data_pedido DATE,
valor_pedido DECIMAL(10,2),
cliente_id INT,
foreign key(cliente_id) references cliente(id)
);
